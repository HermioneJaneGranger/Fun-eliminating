#include "stdafx.h"

#include <string.h>


int index2(char *s, char *t, int pos)

{//返回子串t在主串s中第pos个字符之后的位置。若不存在，返回0
    
int i, j;
   
 i = pos;
    
j = 0;
    
while(i<sizeof(s)/sizeof(s[0]) && j<sizeof(t)/sizeof(t[0]))
    
{
        
if (s[i] == t[j]) 
        
{    
            
++i;
            
++j;
       
 }
       
 else
        
{
            
i = i-j+1;
            
j = 0;
        
}
   
}
    
if (j > sizeof(t)/sizeof(t[0]))
        
return i -sizeof(t)/sizeof(t[0]);
    
else return 0;

}


int _tmain(int argc, _TCHAR* argv[])

{
	
char *s = "abcdaabcd";
	
char *t = "daab";
	
int pos = 0;
	
Index2(s, t, pos);
	
return 0;

}
